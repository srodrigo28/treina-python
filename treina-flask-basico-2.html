app.py 
from flask import Flask, render_template, request, redirect
import mysql.connector

app = Flask(__name__)

# Configura칞칚o do Banco (Ajuste a senha)
def get_db_connection():
    return mysql.connector.connect(
        host="localhost",
        user="root",
        password="",
        database="flask_tarefas"
    )

# ROTA PRINCIPAL: Listar com Filtro e Busca
@app.route('/')
def index():
    conn = get_db_connection()
    cursor = conn.cursor(dictionary=True)
    
    # Captura filtros da URL (?busca=x&categoria=y)
    busca = request.args.get('busca')
    filtro_cat = request.args.get('categoria')

    sql = "SELECT * FROM tarefas WHERE 1=1"
    params = []

    if busca:
        sql += " AND descricao LIKE %s"
        params.append(f"%{busca}%")
    
    if filtro_cat:
        sql += " AND categoria = %s"
        params.append(filtro_cat)

    sql += " ORDER BY id DESC"
    
    cursor.execute(sql, params)
    tarefas = cursor.fetchall()
    
    conn.close()
    return render_template('index.html', tarefas=tarefas)

# ADICIONAR TAREFA
@app.route('/add', methods=['POST'])
def add():
    desc = request.form.get('descricao')
    cat = request.form.get('categoria')
    
    if desc:
        conn = get_db_connection()
        cursor = conn.cursor()
        cursor.execute("INSERT INTO tarefas (descricao, categoria) VALUES (%s, %s)", (desc, cat))
        conn.commit()
        conn.close()
    
    return redirect('/')

# MARCAR COMO FEITO / DESFEITO
@app.route('/toggle/')
def toggle(id):
    conn = get_db_connection()
    cursor = conn.cursor()
    # Inverte o valor booleano
    cursor.execute("UPDATE tarefas SET feito = NOT feito WHERE id = %s", (id,))
    conn.commit()
    conn.close()
    return redirect('/')

# DELETAR TAREFA
@app.route('/delete/')
def delete(id):
    conn = get_db_connection()
    cursor = conn.cursor()
    cursor.execute("DELETE FROM tarefas WHERE id = %s", (id,))
    conn.commit()
    conn.close()
    return redirect('/')

if __name__ == '__main__':
    app.run(debug=True)
